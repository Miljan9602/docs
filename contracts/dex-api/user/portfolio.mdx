---
title: "User Portfolio"
description: "Portfolio overview including historical value, token holdings, and total USD value"
api: "GET https://sei-api.dragonswap.app/api/v1/user/{address}/portfolio"
---

### Path Parameters

<ParamField path="address" type="string" required>
  User wallet address
</ParamField>

### Response

<ResponseField name="status" type="string">
  `"ok"` on success
</ResponseField>

<ResponseField name="meta" type="object">
  Metadata including `native_token_usd_price` and `change` object with `hourly`, `daily`, `weekly`, and `monthly` percentage changes.
</ResponseField>

<ResponseField name="portfolio_history" type="array">
  Historical snapshots of the user's portfolio. Each entry contains `timestamp` (Unix seconds), `liquidity_usd`, and `farm_balance_usd`.
</ResponseField>

<ResponseField name="tokens" type="array">
  Token holdings. Each entry includes `address`, `name`, `symbol`, `usd_price`, `decimals`, `change` object (with `hourly`, `daily`, `weekly`, `monthly`), and `user_balance`.
</ResponseField>

<ResponseField name="total_usd_value" type="number">
  Total portfolio value in USD.
</ResponseField>

<ResponseField name="native_token" type="object">
  The user's native token (SEI) holding with `balance` and `usd_price`.
</ResponseField>
