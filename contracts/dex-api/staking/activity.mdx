</ParamField>

<ParamField path="user" type="string" required>
  User wallet address
</ParamField>

<ResponseField name="status" type="string">
  Response status indicator
</ResponseField>

<ResponseField name="data" type="object">

    </ResponseField>

    <ResponseField name="staking_activity" type="object[]">

    </ResponseField>

    <ResponseField name="marketplace_activity" type="object[]">
      Array of marketplace action records
    </ResponseField>

    <ResponseField name="staking_bonus" type="object[]">
      Array of staking bonus records
    </ResponseField>
  </Expandable>
</ResponseField>
