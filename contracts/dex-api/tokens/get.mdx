---
title: "Get Token"
description: "Returns details for a single token"
api: "GET https://sei-api.dragonswap.app/api/v1/tokens/{address}"
---

## Path Parameters

<ParamField path="address" type="string" required default="0xe30fedd158a2e3b13e9badaeabafc5516e95e8c7">
  Token contract address.
</ParamField>

## Response

<ResponseField name="status" type="string">
  Request status indicator.
</ResponseField>

<ResponseField name="token" type="object">
  Token details object.

  <Expandable title="Token object">
    <ResponseField name="address" type="string">
      Token contract address.
    </ResponseField>
    <ResponseField name="name" type="string">
      Token name.
    </ResponseField>
    <ResponseField name="symbol" type="string">
      Token ticker symbol.
    </ResponseField>
    <ResponseField name="usd_price" type="number">
      Current USD price.
    </ResponseField>
    <ResponseField name="decimals" type="number">
      Token decimal places.
    </ResponseField>
    <ResponseField name="liquidity" type="number">
      Total liquidity in USD.
    </ResponseField>
    <ResponseField name="daily_volume" type="number">
      Trading volume over the last 24 hours in USD.
    </ResponseField>
    <ResponseField name="daily_transactions" type="number">
      Number of transactions over the last 24 hours.
    </ResponseField>
    <ResponseField name="description" type="string|null">
      Token description, if available.
    </ResponseField>
  </Expandable>
</ResponseField>
