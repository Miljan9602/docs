---
title: "Pool Activity"
description: "Paginated pool activity using block number cursors"
api: "GET https://sei-api.dragonswap.app/api/v1/pools/{address}/activity"
---

## Path Parameters

<ParamField path="address" type="string" required default="0x4cea29353508b81b6efbfb8d089efe808e8fc991">
  Pool contract address.
</ParamField>

## Query Parameters

<ParamField query="swaps_from_block" type="number" default="0">
  Block number cursor for swap events. Pass the `block_number` from the last swap result to fetch the next page.
</ParamField>

<ParamField query="mints_from_block" type="number" default="0">
  Block number cursor for mint (add liquidity) events. Pass the `block_number` from the last mint result to fetch the next page.
</ParamField>

<ParamField query="burns_from_block" type="number" default="0">
  Block number cursor for burn (remove liquidity) events. Pass the `block_number` from the last burn result to fetch the next page.
</ParamField>

<Info>
  Results are limited to 500 per request. Use the `block_number` from the last result as the cursor for the next page.
</Info>

## Response Fields

<ResponseField name="status" type="string">
  Request status indicator.
</ResponseField>

<ResponseField name="swaps" type="array">
  Array of swap activity objects.

  <Expandable title="Swap object">
    <ResponseField name="pool_address" type="string">
      Address of the pool where the swap occurred.
    </ResponseField>
    <ResponseField name="first_token_address" type="string">
      Address of the first token in the swap.
    </ResponseField>
    <ResponseField name="second_token_address" type="string">
      Address of the second token in the swap.
    </ResponseField>
    <ResponseField name="first_amount_token" type="number">
      Amount of the first token swapped.
    </ResponseField>
    <ResponseField name="second_amount_token" type="number">
      Amount of the second token swapped.
    </ResponseField>
    <ResponseField name="block_number" type="number">
      Block number in which the transaction was included.
    </ResponseField>
    <ResponseField name="tx_hash" type="string">
      Transaction hash.
    </ResponseField>
    <ResponseField name="from" type="string">
      Address that initiated the swap.
    </ResponseField>
    <ResponseField name="total_value" type="number">
      Total value of the swap in USD.
    </ResponseField>
    <ResponseField name="type" type="string">
      Transaction type: `Swap`.
    </ResponseField>
    <ResponseField name="action" type="string">
      Swap direction: `Buy` or `Sell`.
    </ResponseField>
    <ResponseField name="created_at" type="string">
      Timestamp of the transaction.
    </ResponseField>
    <ResponseField name="pool_type" type="string">
      Pool type: `V3_POOL` or `V2_POOL`.
    </ResponseField>
  </Expandable>
</ResponseField>
